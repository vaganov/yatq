160 uid = _next_uid++;

243 void purge() {
244#ifndef YATQ_DISABLE_LOGGING
245 static auto logger = log4cxx::Logger::getLogger("yatq.timer_queue");
246#endif
247
248 std::size_t canceled_timers;
249 {
250 std::lock_guard<std::mutex> guard(_lock);
251 auto total_jobs = _jobs.size();
252 auto total_timers = _heap.size();
253 canceled_timers = total_timers - total_jobs;
254 if (total_timers > total_jobs) {
255 std::vector<HeapEntry> heap(total_jobs);
256 auto i = heap.begin();
257 for (auto&& heap_entry: _heap) {
258 if (_jobs.contains(heap_entry.uid)) {
259 *i++ = std::move(heap_entry);
260 }
261 else {
262 LOG4CXX_DEBUG(logger, std::format("Timer uid={} has been canceled", heap_entry.uid));
263 }
264 }
265 std::make_heap(heap.begin(), heap.end(), TimerQueue::heap_cmp);
266 _heap.swap(heap);
267 }
268 }
269 // NB: 'demux()' never waits on a canceled timer => no need to notify
270 LOG4CXX_DEBUG(logger, std::format("Purged {} canceled timers", canceled_timers));
271 }

287 void demux() {
288#ifndef YATQ_DISABLE_LOGGING
289 static auto logger = log4cxx::Logger::getLogger("yatq.timer_queue");
290#endif
291
292 SET_THREAD_TAG("timer_queue");
293 LOG4CXX_INFO(logger, "Start");
294
295 std::unique_lock<std::mutex> guard(_lock);
296 while (_running) {
297 bool deadline_expired = false;
298 while (!_heap.empty()) {
299 auto current_uid = _heap[0].uid;
300 auto i = _jobs.find(current_uid);
301 if (i == _jobs.end()) {
302 LOG4CXX_DEBUG(logger, std::format("Timer uid={} has been canceled", current_uid));
303 std::pop_heap(_heap.begin(), _heap.end(), TimerQueue::heap_cmp);
304 _heap.pop_back();
305 deadline_expired = false;
306 continue;
307 }
308 if (!deadline_expired) {
309 auto now = Clock::now(); // NB: system call => context switch
310 deadline_expired = (_heap[0].deadline <= now);
311 }
312 if (deadline_expired) {
313 LOG4CXX_DEBUG(logger, std::format("Executing timer uid={}", current_uid));
314 auto node = _jobs.extract(i);
315 std::pop_heap(_heap.begin(), _heap.end(), TimerQueue::heap_cmp);
316 _heap.pop_back();
317 auto map_entry = std::move(node.mapped());
318
319 guard.unlock();
320#ifndef YATQ_DISABLE_FUTURES
321 auto future =
322#endif
323 _executor->execute(std::move(map_entry.job));
324 guard.lock();
325
326#ifndef YATQ_DISABLE_FUTURES
327 future.then( // future chaining -- this is why we use 'boost::future' instead of 'std::future'
328 boost::launch::sync, // FIXME: does not match the concept
329 [promise = std::move(map_entry.promise)]
330 (Executor::Future future) mutable
331 { internal::get_and_set_value<result_type>(std::move(future), std::move(promise)); }
332 );
333#endif
334 deadline_expired = false;
335 }
336 else {
337 // NB: without this explicit cast duration type may be deduced incorrectly
338 // on Linux this leads to waiting for a random time point
339 std::chrono::time_point<Clock, typename Clock::duration> deadline = _heap[0].deadline;
340 LOG4CXX_TRACE(logger, std::format("Wait until {}", utils::time_point_to_string(deadline)));
341 bool notified = _cond.wait_until(
342 guard,
343 deadline,
344 [this, current_uid] () {
345 return !_jobs.contains(current_uid) || (_heap[0].uid != current_uid) || !_running;
346 }
347 );
348 LOG4CXX_TRACE(logger, "Wake-up");
349 if (!_running) {
350 LOG4CXX_WARN(logger, "Stopping timer queue with unprocessed timers");
351 return;
352 }
353 if (!notified) { // => timeout
354 deadline_expired = true;
355 }
356 }
357 }
358 LOG4CXX_TRACE(logger, "Wait");
359 _cond.wait(guard, [this] () { return !_heap.empty() || !_running; });
360 LOG4CXX_TRACE(logger, "Wake-up");
361 }
362
363 LOG4CXX_INFO(logger, "Stop");
364 }
